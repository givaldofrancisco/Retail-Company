sql=WITH quarterly_metrics AS (
    SELECT
        CASE
            WHEN DATE(o.created_at) >= DATE_TRUNC(CURRENT_DATE(), QUARTER) AND DATE(o.created_at) < DATE_ADD(DATE_TRUNC(CURRENT_DATE(), QUARTER), INTERVAL 1 QUARTER) THEN 'This Quarter'
            WHEN DATE(o.created_at) >= DATE_SUB(DATE_TRUNC(CURRENT_DATE(), QUARTER), INTERVAL 1 QUARTER) AND DATE(o.created_at) < DATE_TRUNC(CURRENT_DATE(), QUARTER) THEN 'Last Quarter'
            ELSE 'Other' -- Should not be reached with the outer WHERE clause
        END AS quarter_period,
        oi.sale_price,
        o.order_id
    FROM `bigquery-public-data.thelook_ecommerce.order_items` oi
    JOIN `bigquery-public-data.thelook_ecommerce.orders` o
    ON
        oi.order_id = o.order_id
    WHERE
        DATE(o.created_at) >= DATE_SUB(DATE_TRUNC(CURRENT_DATE(), QUARTER), INTERVAL 1 QUARTER)
        AND DATE(o.created_at) < DATE_ADD(DATE_TRUNC(CURRENT_DATE(), QUARTER), INTERVAL 1 QUARTER)
)
SELECT
    quarter_period,
    ROUND(SUM(sale_price), 2) AS total_revenue,
    COUNT(DISTINCT order_id) AS total_orders
FROM quarterly_metrics
WHERE
    quarter_period IN ('This Quarter', 'Last Quarter')
GROUP BY
    quarter_period
ORDER BY
    quarter_period ASC LIMIT 200
