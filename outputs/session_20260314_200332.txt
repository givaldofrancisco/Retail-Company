This is achieved by `DATE(o.created_at) >= DATE_SUB(CURRENT_DATE(), INTERVAL 2 MONTH)`. This ensures we capture data for the current month and the preceding month(s) for comparison.

WITH monthly AS (
    SELECT
        FORMAT_DATE('%Y-%m', DATE(o.created_at)) AS month,
        SUM(oi.sale_price) AS revenue
    FROM `bigquery-public-data.thelook_ecommerce.order_items` oi
    JOIN `bigquery-public-data.thelook_ecommerce.orders` o ON oi.order_id = o.order_id
    WHERE DATE(o.created_at) >= DATE_SUB(CURRENT_DATE(), INTERVAL 2 MONTH)
    GROUP BY month
)
SELECT
    month,
    ROUND(revenue, 2) AS revenue
FROM monthly
ORDER BY month
